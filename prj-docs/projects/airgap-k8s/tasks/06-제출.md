```bash
find manual -type f -name '*.md' | sort
find ops -type f \
  -not -path '*/.terraform/*' \
  -not -name '*.tfstate' \
  -not -name '*.tfstate.backup' \
  -not -name 'terraform.tfvars' \
  -not -name 'tfplan' \
  -not -name '*.pem' \
  | sort
```

```bash
rsync -av assets/offline-assets/ delivery/offline-assets/
tar -czf dist/offline-assets.tar.gz -C delivery offline-assets
```
